
### Context

-   Create a context

```jsx
export const Context = React.createContext();
export const Provider = ({ children }) => (
    <Context.Provider value={{ id, name }}>{children}</Context.Provider>
);
```

-   Usage

```jsx
// App.js
<Provider>
    <App />
</Provider>
```

```jsx
import { Context } from './appContext';
const { id, name } = useContext(Context);
```
